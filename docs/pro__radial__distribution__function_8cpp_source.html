46 void ProRadialDistributionFunction::run_impl(
47  const int index)
48 {
49  /* NOTE:
50  Although 'atom' is used for variable names,
51  RDF of something other than atoms can be computed too.
52  */
53  auto el_atoms = generators[index]->get_element("Targets");
54 
55  el_atoms->required({"x", "y", "z", "id"});
56 
57  auto &atoms = el_atoms->get_data();
58 
59  bool special_bonds_exist = el_atoms->optional("special-bonds");
60 
61  auto el_box = generators[index]->get_element("Box");
62 
63  el_box->required({"lo_x", "lo_y", "lo_z", "hi_x", "hi_y", "hi_z"});
64 
65  auto &box = el_box->get_data();
66 
67  ArrayXXd rs;
68  el_atoms->array2d(rs, {"x", "y", "z"});
69 
70  ArrayXd length(3);
71  length << box["hi_x"].get<double>() - box["lo_x"].get<double>(),
72  box["hi_y"].get<double>() - box["lo_y"].get<double>(),
73  box["hi_z"].get<double>() - box["lo_z"].get<double>();
74 
75  ArrayXd limits = beyond_half ? length : 0.5 * length;
76 
77  auto r_max = bin_from_r ?
78  n_bins * bin_width : (n_bins-0.5) * bin_width;
79  auto r2_max = r_max*r_max;
80 
81  for (int i = 0; i != 3; ++i)
82  {
83  if (limits(i) < r_max)
84  {
85  ut::warning(
86  "Box length is too short in " + Str("xyz").substr(i, 1));
87  }
88  }
89 
90  std::pair<int,int> shift_x = box.value("periodic_x", false) ?
91  std::make_pair(-1, 1) : std::make_pair(0, 0);
92  std::pair<int,int> shift_y = box.value("periodic_y", false) ?
93  std::make_pair(-1, 1) : std::make_pair(0, 0);
94  std::pair<int,int> shift_z = box.value("periodic_z", false) ?
95  std::make_pair(-1, 1) : std::make_pair(0, 0);
96 
97  auto reciprocal_width = 1.0 / bin_width;
98 
99  auto n_atoms = atoms.size();
100 
101  number_traj[index] = n_atoms;
102  volume_traj[index] = length.prod();
103  counts_traj[index] = ArrayXi::Zero(n_bins);
104 
105  auto &counts = counts_traj[index];
106 
107  for (int i = 0; i != n_atoms; ++i)
108  {
109  auto sbs_i = special_bonds_exist ?
110  atoms[i]["special-bonds"].get<Set<int>>() : Set<int>();
111 
112  auto r_i = rs.row(i);
113 
114  for (int j = i+1; j != n_atoms; ++j)
115  {
116  if (!sbs_i.empty() && sbs_i.find(
117  atoms[j]["id"].get<int>()) != sbs_i.end()) continue;
118 
119  auto dr_original = rs.row(j) - r_i;
120 
121  for (int ix = shift_x.first; ix <= shift_x.second; ++ix)
122  {
123  auto dx = dr_original(0) + ix*length(0);
124 
125  if (limits(0) < abs(dx)) continue;
126 
127  for (int iy = shift_y.first; iy <= shift_y.second; ++iy)
128  {
129  auto dy = dr_original(1) + iy*length(1);
130 
131  if (limits(1) < abs(dy)) continue;
132 
133  for (int iz = shift_z.first; iz <= shift_x.second; ++iz)
134  {
135  auto dz = dr_original(2) + iz*length(2);
136 
137  if (limits(2) < abs(dz)) continue;
138 
139  auto r2 = dx*dx + dy*dy + dz*dz;
140 
141  if (r2_max <= r2) continue;
142 
143  auto r = sqrt(r2);
144 
145  auto r_index = bin_from_r ?
146  floor(r*reciprocal_width) : round(r*reciprocal_width);
147 
148 
149  /* NOTE:
150  Adding 2 (not 1) is for taking both directions
151  (i -> j & j -> i) into consideration at once.
152  */
153  counts(r_index) += 2;
154 
155  }
156  }
157  }
158  }
159  }
160 }

173 void ProRadialDistributionFunction::finish()
174 {
175  ArrayXd shell_volume(n_bins);
176 
177  double ball_coeff = 4.0 * M_PI / 3.0;
178 
179  for (int i = 0; i != n_bins; ++i)
180  {
181  auto r_inner = bin_from_r ?
182  i * bin_width : std::max(0.0, (i-0.5) * bin_width);
183  auto r_outer = bin_from_r ?
184  (i+1) * bin_width : (i+0.5) * bin_width;
185 
186  shell_volume(i) = ball_coeff * (pow(r_outer, 3) - pow(r_inner, 3));
187  }
188 
189  int number_sum = 0;
190  double volume_sum = 0.0;
191  ArrayXi counts_sum = ArrayXi::Zero(n_bins);
192 
193  rdf_traj.reserve(n_generators);
194 
195  for (int i = 0; i != n_generators; ++i)
196  {
197  auto &number_tmp = number_traj[i];
198  auto &volume_tmp = volume_traj[i];
199  auto &counts_tmp = counts_traj[i];
200 
201  auto density_tmp = number_tmp / volume_tmp;
202  auto number_distribution_tmp
203  = counts_tmp.cast<double>() / double(number_tmp);
204 
205  rdf_traj.push_back(
206  number_distribution_tmp / (density_tmp*shell_volume));
207 
208  number_sum += number_tmp;
209  volume_sum += volume_tmp;
210  counts_sum += counts_tmp;
211  }
212 
213  auto density = number_sum / volume_sum;
214  auto number_distribution
215  = counts_sum.cast<double>() / double(number_sum);
216 
217  rdf = number_distribution / (density*shell_volume);
218 
219  number_traj.clear();
220  number_traj.shrink_to_fit();
221  volume_traj.clear();
222  volume_traj.shrink_to_fit();
223  counts_traj.clear();
224  counts_traj.shrink_to_fit();
225 }
